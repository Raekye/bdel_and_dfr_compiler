def malloc(var n) {
	var i = asm {
		%literal 0 r0;
		%unheap r0 r13;
	};
	var ret = 0 - 1;
	while (i < 1024) {
		var x = heap_get(i);
		if (x == 0) {
			i = i + 2;
			var j = 0;
			var ok = 1;
			while (j < n) {
				if (heap_get(i + j) == 0) {
					0;
				} else {
					ok = 0;
					break;
				}
				j = j + 1;
			}
			if (ok == 1) {
				ret = i;
				break;
			} else {
				i = i + j;
			}
		} else {
			i = i + 1 + heap_get(i + 1);
		}
	}
	if (i > 1023) {
		error(1);
	}
	ret;
}

def free(var address) {
	var n = heap_get(address - 1);
	var i = address - 2;
	while (i < address + n) {
		heap_set(i, 0);
	}
}
