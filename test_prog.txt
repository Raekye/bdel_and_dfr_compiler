def fib(var x);

def main() {
	var a = 20;
	fib(a);
}

def fib(var x) {
	if (x == 0) {
		0;
	} else {
		if (x == 1) {
			1;
		} else {
			fib(x - 1) + fib(x - 2);
		};
	};
}
